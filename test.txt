def calcularFac(num):
    num_aux = 0
    if (num < 1):
        num_aux = -1
    else:
        num_aux = num * (calcularFac(num--1))
    return num_aux

def promedio(cualquier_arreglo):
    tam = len(cualquier_arreglo)
